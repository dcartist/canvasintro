server.js — Entry point. Configures ltijs, registers the Canvas platform, and starts the Express server. 
import Lti       from 'ltijs'
import { launch }  from './routes/launch.js'
import { deeplink, submitDeeplink }from './routes/deeplink.js'
import 'dotenv/config'

// 1. Configure ltijs with your database and secret
Lti.setup(
  process.env.LTI_SECRET,
  { url: process.env.MONGO_URL },
  {
    // v5 option names: appRoute / loginRoute / keysetRoute
    appRoute:    '/',
    loginRoute:  '/login',
    keysetRoute: '/keys',
    // cookies MUST be cross-domain safe — Canvas loads your tool in an iframe
    cookies: {
      secure:   true,         // required when tool and Canvas are on different domains
      sameSite: 'None'        // required for cross-domain iframe cookie to be sent
    },
    devMode: process.env.NODE_ENV !== 'production' // disables HTTPS check locally
  }
)

// 2. Handle regular launch (assignment page, nav, etc.)
//    onConnect must be registered BEFORE deploy()
Lti.onConnect((token, req, res) => {
  launch(token, req, res)
})

// 3. Handle Deep Linking (editor_button placement)
Lti.onDeepLinking((token, req, res) => {
  deeplink(token, req, res)
})

// 4. Boot — registerPlatform must come AFTER deploy()
await Lti.deploy({ port: 3000 })

// Register the deep linking submit route on Lti's Express instance
Lti.app.post('/deeplink/submit', submitDeeplink)

// 5. Register your Canvas instance (only stored once; ltijs skips
//    duplicate registrations on subsequent restarts)
await Lti.registerPlatform({
  url:                    process.env.CANVAS_URL,
  name:                   'Canvas',
  clientId:               process.env.CANVAS_CLIENT_ID,
  authenticationEndpoint: process.env.CANVAS_AUTH_URL,
  accesstokenEndpoint:    process.env.CANVAS_TOKEN_URL,
  authConfig: { method: 'JWK_SET', key: process.env.CANVAS_JWK_URL }
})

console.log('LTI tool running on port 3000')

routes/launch.js — Handles regular LTI launches (course navigation, assignment pages). Extracts the JWT token and injects context into the page. 
import path from 'path'
import { readFileSync } from 'fs'

export function launch(token, req, res) {
  // Extract user + course info from the JWT token
  const context = {
    user: {
      id:    token.user,
      name:  token.userInfo?.name ?? 'Unknown',
      email: token.userInfo?.email ?? ''
    },
    course: {
      id:    token.platformContext?.context?.id,
      title: token.platformContext?.context?.title ?? 'Unknown course'
    },
    roles: token.platformContext?.roles ?? []
  }

  // Read the HTML template and inject context as a script tag
  let html = readFileSync('public/index.html', 'utf8')
  html = html.replace(
    '</head>',
    `<script>window.__LTI_CONTEXT__ = ${JSON.stringify(context)};</script></head>`
  )

  res.setHeader('Content-Type', 'text/html')
  res.send(html)
}
routes/deeplink.js — Deep Linking is a two-step flow. deeplink() (called by onDeepLinking) serves your selection UI when Canvas first opens the modal. submitDeeplink() is a separate POST handler that receives the user's choice and returns the Deep Linking response to Canvas. 
import Lti from 'ltijs'

/**
 * Step 1 — onDeepLinking fires when Canvas first opens the modal.
 * Serve your selection UI here. Do NOT try to read req.body yet —
 * the user hasn't chosen anything.
 */
export function deeplink(token, req, res) {
  // Serve the selection UI. ltijs passes the ltik as a query param (?ltik=...)
  // Your HTML form should include it as a hidden field so submitDeeplink can
  // validate the session when the user submits their content choice.
  const ltik = res.locals.ltik
  res.send(`<!DOCTYPE html>
<html lang="en"><body>
  <h1>Select content to embed</h1>
  <form method="POST" action="/deeplink/submit">
    <input type="hidden" name="ltik" value="${ltik}">
    <select name="selectedId">
      <option value="item-1">Item 1</option>
      <option value="item-2">Item 2</option>
    </select>
    <button type="submit">Insert</button>
  </form>
</body></html>`)
}

/**
 * Step 2 — A separate POST route receives the user's selection
 * and returns the Deep Linking response to Canvas.
 * Register this on Lti.app AFTER deploy():
 *   Lti.app.post('/deeplink/submit', submitDeeplink)
 */
export async function submitDeeplink(req, res) {
  // ltik is passed back by your UI as a query param or Authorization header
  const items = [
    {
      type: 'html',
      html: `<iframe
        src="https://yourtool.com/embed?id=${req.body.selectedId}"
        width="100%" height="400"
        title="My LTI Tool embed"></iframe>`
    }
  ]

  const message = await Lti.DeepLinking.createDeepLinkingMessage(
    res.locals.token,  // token is on res.locals after ltijs validates the ltik
    items,
    { message: 'Content added successfully!' }
  )

  res.send(message)
}

How the launch flow works

Every time Canvas opens your tool, it follows the LTI 1.3 OIDC launch sequence:

Launch sequence — step by step
StepWhoWhat happens
1. InitiationCanvasCanvas POSTs to your /login endpoint with a login hint and target URI
2. OIDC redirectYour serverYou validate the hint and redirect the browser back to Canvas's auth endpoint
3. JWT launchCanvasCanvas POSTs a signed JWT (id_token) back to your app route (/) via ltijs internals
4. Token verifiedltijsltijs verifies the JWT using Canvas's JWKS, checks nonce, builds the token object
5. onConnect firesYour codeYour launch.js handler runs — user/course data is available in token
6. UI loadsBrowserYour tool's HTML is returned and displayed inside a Canvas iframe
ℹ ltijs handles steps 2–4 automatically. You only write code for step 5 (onConnect / onDeepLinking) and step 6 (your HTML/UI).

Section 02

What is LTI?

LTI (Learning Tools Interoperability) lets you embed external tools inside Canvas. Your tool acts as an external web app that Canvas launches via a signed JWT handshake.

Platform

Canvas LMS — the host that initiates launches and manages users and courses.

Tool

Your external web app, loaded in an iframe or modal inside Canvas.

Launch

The moment Canvas opens your tool, passing user and course context via a signed JWT.

ℹ Canvas uses LTI 1.3 — the current standard. LTI 1.1 is deprecated. Always build for 1.3.

Key Endpoints Your App Needs

Routes
// Canvas will call these on your server (matches server.js config)
/login  ← OIDC initiation — Canvas starts every launch here (loginRoute)
/       ← App route — onConnect fires here after a successful launch (appRoute)
/keys   ← JWK keyset — Canvas fetches your public key here (keysetRoute)

Section 05

Custom Modals via LTI

Three approaches to show a modal when your LTI tool is loaded inside Canvas.

Option A

Canvas PostMessage API

Trigger a native Canvas tray or modal from inside your iframe using window.parent.postMessage.

JavaScript
window.parent.postMessage(
  {
    subject: 'requestFullWindowLaunch',
    data: {
      url:       'https://yourtool.com/launch',
      placement: 'course_navigation', // required — match your tool's placement
      launchType: 'popup',              // 'same_window' | 'new_window' | 'popup'
      launchOptions: { width: 800, height: 600 }
    }
  },
  '*'
)
Option B

Modal Inside Your Iframe

Build your own modal UI entirely within the iframe. Include ARIA roles so it is accessible.

HTML
<div role="dialog" aria-modal="true"
     aria-labelledby="modal-title"
     style="display:none; position:fixed; inset:0;">
  <h2 id="modal-title">My LTI Tool</h2>
  <button aria-label="Close dialog">Close</button>
</div>
Option C — Most Native

editor_button + Deep Linking Recommended

Canvas opens your tool in a native modal from the RCE toolbar. Return content back to Canvas via a Deep Linking response (see deeplink.js in the App Structure tab).

Section 06

Modal Without LTI

You can add a custom modal to Canvas assignment pages without any LTI setup.

Option 1 — Admin Required

Custom JavaScript via Theme Editor Admin

Upload a JS file in Admin → Themes → Custom JS. Runs on every Canvas page globally.

JavaScript
function injectModal() {
  if (!window.location.pathname.match(/\/assignments\/\d+/)) return;

  const wrap = document.createElement('div');
  wrap.innerHTML = `
    <div id="modal" role="dialog" aria-modal="true"
         aria-labelledby="modal-title"
         style="display:none; position:fixed; inset:0;
                background:rgba(0,0,0,0.5); z-index:9999;">
      <div style="background:#fff; margin:8% auto;
                  padding:2rem; width:560px; border-radius:8px;">
        <button id="modal-close" aria-label="Close">✕</button>
        <h2 id="modal-title">My Custom Tool</h2>
      </div>
    </div>
  `;
  document.body.appendChild(wrap);

  const btn = document.createElement('button');
  btn.textContent = 'Open Tool';
  const area = document.querySelector('.submit_assignment_link');
  if (area) area.after(btn);

  btn.addEventListener('click', () => {
    document.getElementById('modal').style.display = 'block';
  });
  document.getElementById('modal-close').addEventListener('click', () => {
    document.getElementById('modal').style.display = 'none';
  });
}

if (document.readyState === 'loading')
  document.addEventListener('DOMContentLoaded', injectModal);
else
  injectModal();
Option 2 — No Admin Needed

Tampermonkey Userscript Per-User

Install the Tampermonkey browser extension and paste the script above with this header.

JavaScript
// ==UserScript==
// @name    Canvas Assignment Modal
// @match   https://*.instructure.com/courses/*/assignments/*
// @grant   none
// ==/UserScript==

Section 10

Two Ways to Build an LTI Tool

You don't always need the full ltijs setup. Here are both approaches — pick the one that fits what you're building.

Lightweight — single file

A minimal PHP or Node.js file. No database, no session library. You verify the JWT yourself using a small library.

  • Runs on any shared host (cPanel, Dreamhost, etc.)
  • No MongoDB or database needed
  • One or two files total
  • Good for read-only tools — show info, embed content
  • Not ideal for storing grades or user data
Full server — ltijs + Node.js

The complete setup from Tab 01. ltijs handles OIDC, sessions, nonces, and Deep Linking automatically.

  • Needs a persistent Node.js server + MongoDB
  • Handles complex flows automatically
  • Supports grade passback (LTI Advantage)
  • Good for tools that store data or send grades back
  • More setup but far more capable

Which should you use?

Lightweight approach — PHP (single file)

Upload one file to any PHP host. Install one dependency with Composer. Point Canvas at its URL. Done.

bash
composer require firebase/php-jwt
php — lti.php
<?php
// lti.php — upload to any PHP host
// Set these in your hosting control panel as environment variables,
// or replace the getenv() calls with your values directly.
$CLIENT_ID  = getenv('CANVAS_CLIENT_ID');
$CANVAS_URL = getenv('CANVAS_URL');
$AUTH_URL   = getenv('CANVAS_AUTH_URL');
$JWK_URL    = getenv('CANVAS_JWK_URL');
$TOOL_URL   = getenv('TOOL_URL');   // full URL to this file

require_once __DIR__ . '/vendor/autoload.php';

// ── Step 2: Canvas POSTs the signed JWT here ──────────────────
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['id_token'])) {

    $jwks   = json_decode(file_get_contents($JWK_URL), true);
    $keySet = \Firebase\JWT\JWK::parseKeySet($jwks);

    try {
        $token = \Firebase\JWT\JWT::decode($_POST['id_token'], $keySet);

        $user   = $token->name ?? 'Student';
        $ctx    = $token->{'https://purl.imsglobal.org/spec/lti/claim/context'} ?? null;
        $course = $ctx->title ?? 'Unknown course';

        // ── Your tool UI goes here ──────────────────────────────────
        echo "<!DOCTYPE html><html lang='en'><body>
          <h1>Hello, " . htmlspecialchars($user) . "!</h1>
          <p>Course: " . htmlspecialchars($course) . "</p>
        </body></html>";

    } catch (Exception $e) {
        http_response_code(403);
        echo 'Launch failed: ' . htmlspecialchars($e->getMessage());
    }
    exit;
}

// ── Step 1: Canvas calls this to start the launch ─────────────
if (isset($_GET['login_hint'])) {
    session_start();
    $nonce = bin2hex(random_bytes(16));
    $_SESSION['lti_nonce'] = $nonce;

    $params = http_build_query([
        'scope'            => 'openid',
        'response_type'    => 'id_token',
        'response_mode'    => 'form_post',
        'prompt'           => 'none',
        'client_id'        => $CLIENT_ID,
        'redirect_uri'     => $TOOL_URL,
        'login_hint'       => $_GET['login_hint'],
        'lti_message_hint' => $_GET['lti_message_hint'] ?? '',
        'state'            => bin2hex(random_bytes(16)),
        'nonce'            => $nonce,
    ]);

    header('Location: ' . $AUTH_URL . '?' . $params);
    exit;
}

echo 'LTI tool ready.';
ℹ Register this in Canvas Developer Keys with Target Link URI and OIDC Initiation URL both pointing to the same file URL (e.g. https://yourdomain.com/lti.php). The file handles both steps based on whether the request is a GET or POST.

Lightweight approach — Node.js without ltijs

Same idea in Node.js. No ltijs, no MongoDB — just express and jose. Runs on any Node.js host.

bash
npm install express jose dotenv
javascript — server.js
import express    from 'express'
import * as jose  from 'jose'
import             'dotenv/config'

const app    = express()
const nonces = new Set()   // in-memory nonce store

app.use(express.urlencoded({ extended: true }))

// ── Step 1: Canvas calls /login to start the launch ───────────
app.get('/login', (req, res) => {
    const nonce = crypto.randomUUID()
    nonces.add(nonce)
    setTimeout(() => nonces.delete(nonce), 60_000) // expire after 60 s

    const params = new URLSearchParams({
        scope:            'openid',
        response_type:    'id_token',
        response_mode:    'form_post',
        prompt:           'none',
        client_id:        process.env.CANVAS_CLIENT_ID,
        redirect_uri:     process.env.TOOL_URL,
        login_hint:       req.query.login_hint,
        lti_message_hint: req.query.lti_message_hint ?? '',
        state:            crypto.randomUUID(),
        nonce,
    })
    res.redirect(process.env.CANVAS_AUTH_URL + '?' + params)
})

// ── Step 2: Canvas POSTs the signed JWT here ──────────────────
app.post('/', async (req, res) => {
    try {
        const JWKS = jose.createRemoteJWKSet(new URL(process.env.CANVAS_JWK_URL))

        const { payload } = await jose.jwtVerify(req.body.id_token, JWKS, {
            issuer:   process.env.CANVAS_URL,
            audience: process.env.CANVAS_CLIENT_ID,
        })

        // Verify nonce to prevent replay attacks
        if (!nonces.has(payload.nonce)) return res.status(403).send('Invalid nonce')
        nonces.delete(payload.nonce)

        const user   = payload.name ?? 'Student'
        const course = payload['https://purl.imsglobal.org/spec/lti/claim/context']?.title ?? 'Unknown'
        const roles  = payload['https://purl.imsglobal.org/spec/lti/claim/roles'] ?? []

        // ── Your tool UI goes here ──────────────────────────────────
        res.send(`<!DOCTYPE html>
<html lang="en"><body>
  <h1>Hello, ${user}!</h1>
  <p>Course: ${course}</p>
  <p>Role: ${roles.map(r => r.split('#').pop()).join(', ')}</p>
</body></html>`)

    } catch (err) {
        res.status(403).send('Launch failed: ' + err.message)
    }
})

app.listen(3000, () => console.log('LTI tool ready on port 3000'))
⚠ The in-memory nonce store resets on every server restart. For production use a persistent store like Redis or a database table to prevent replay attacks across restarts.

What Is All This? Plain English.

Everything in this guide explained simply — like you're 10 years old and just want to know what's going on.

🏢 What is Canvas?

Canvas is like a giant school building on the internet. Students and teachers use it every day — to turn in homework, read lessons, see grades, and talk to each other. It lives on a website, not on your computer.

🚪 What is an LTI plugin?

Imagine the school building already has classrooms, a gym, and a cafeteria. But you want to add a brand new science lab that wasn't there before. An LTI plugin is that science lab — a new room you built yourself and attached to the building. When a student clicks your room's door in Canvas, it opens right there on the page inside a little window.

The important thing: your room lives somewhere else — on your own computer or a rented computer on the internet. Canvas just knows the address of your room and shows it inside a frame.

💻 What is "a server"?

A server is just a computer that's always switched on and connected to the internet, so anyone can visit it any time. When you rent one, you put your code on it and it runs 24 hours a day. Canvas talks to your server every time a student opens your plugin.

🪪 The hall pass (what is a JWT?)

Before Canvas opens your room to a student, it gives your server a sealed hall pass. The hall pass has the student's name, their course, and their role (student or teacher) written inside — and Canvas signs it with a secret stamp so nobody can fake it. Your server checks the stamp, reads the hall pass, and knows exactly who it's talking to.

That hall pass is called a JWT (JSON Web Token). The secret stamp uses math to make it impossible to fake.

🔧 The two ways to build a plugin

🪶 The Light Way
One or two files. Simple hosting. Like a portable classroom you roll in on wheels. Quick to set up. Works on cheap web hosting. Good for showing information to students. Can't save a lot of data by itself.
🏗️ The Full Way
A whole project with a database. Like a real built-in lab with plumbing and electricity. More work to set up but can do everything — remember students, save their work, send grades back to Canvas.

✅ What you actually do — 5 steps

  1. 1 Write your tool. Build a web page or app that does what you want. It can be as simple as one PHP file or as complex as a full Node.js app.
  2. 2 Put it on the internet. Upload your files to a web host so it has a public address — like https://mytool.com. It must have HTTPS (the padlock).
  3. 3 Tell Canvas your address. Go into Canvas admin, create a Developer Key, and paste in your tool's URL. This is like giving Canvas your room's phone number.
  4. 4 Add it to a course. Go to the course settings and install the tool using the ID Canvas gave you. It shows up as a link in the course menu.
  5. 5 Click it in Canvas. When a student clicks your tool, Canvas does the hall pass handshake with your server, and your page loads inside a Canvas window. Done!

❓ Quick questions

Do I need to know how to code?

Yes — at least a little. You need to write a web page (HTML) and a small server script (PHP or JavaScript). The code in this guide gives you a working starting point you can copy and change.

Does my plugin live inside Canvas?

No. Your plugin always lives on your own server. Canvas just opens a window to it — like how a web page can show a YouTube video inside it even though the video lives on YouTube's servers.

Do I need to pay for a server?

For testing, no — you can use a free tunnel tool called ngrok to let Canvas talk to your computer directly. For a real live tool, yes — a basic server costs about $5–10 a month, or you can use free tiers on hosting services.

What if I just want to add a button to Canvas without all this?

If your school gives you admin access, you can add a small JavaScript file through the Canvas Theme Editor. It's quicker but less powerful — it can add buttons and popups but Canvas might break it when they update their code.

What is ltijs?

ltijs is a helper library for Node.js that handles all the complicated parts of the LTI handshake for you — like checking the hall pass, managing sessions, and talking to Canvas. Without it, you have to write that code yourself (which the lightweight approach does).

The one-sentence version: An LTI plugin is a web page you build and host yourself, that Canvas opens in a little window inside a course — and before it opens, Canvas sends your page a secret signed note that says who the student is.
